var $on
var $off
var $interval
var $steps

disable logs
io only outputs

$on =  0x01
$off = 0x00

// cycle: gpio0 - tx - gpio2 - rx

@repeat
$steps = 8

@step
  @0 = $on
  yield
  @1 = $on
  @0 = $off
  yield

  @2 = $on
  @1 = $off
  yield

  @3 = $on
  @2 = $off
  yield

  @3 = $off

  --$steps
jump if $steps to @step

wait 1
jump to @repeat
